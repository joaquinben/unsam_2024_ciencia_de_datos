Filtrar por una región que me interesa

# %>% 

my_data$REGION |> 
  unique() # Es para verificar
[1] 42 40  1 43 41 44
my_data <- my_data |>
  filter(REGION == 44) #es para filtrar.
nrow(my_data)
[1] 2175
